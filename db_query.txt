-- Table: admin
CREATE TABLE IF NOT EXISTS public.admin
(
    id integer NOT NULL,
    first_name character varying(255) COLLATE pg_catalog."default" NOT NULL,
    last_name character varying(255) COLLATE pg_catalog."default" NOT NULL,
    contact bigint,
    email character varying(255) COLLATE pg_catalog."default" NOT NULL,
    hashed_password character varying(255) COLLATE pg_catalog."default" NOT NULL,
    is_deactive boolean NOT NULL DEFAULT false,
    is_delete boolean NOT NULL DEFAULT false,
    is_firstlogin boolean NOT NULL DEFAULT true,
    role_id integer NOT NULL,
    invited_at timestamp without time zone,
    deleted_at timestamp without time zone,
    created_at timestamp without time zone,
    CONSTRAINT admin_pkey PRIMARY KEY (id),
    CONSTRAINT "UQ_386657905f0fdeabc53555beba3" UNIQUE (email),
    CONSTRAINT "FK_fd32421f2d93414e46a8fcfd86b" FOREIGN KEY (role_id)
        REFERENCES public.role (id) MATCH SIMPLE
        ON UPDATE NO ACTION
        ON DELETE CASCADE
)


-- Table: admin_role

CREATE TABLE IF NOT EXISTS public.role
(
    id integer NOT NULL,
    name character varying(50) COLLATE pg_catalog."default" NOT NULL,
    created_at timestamp without time zone NOT NULL DEFAULT now(),
    CONSTRAINT role_pkey PRIMARY KEY (id),
    CONSTRAINT role_name_key UNIQUE (name)
)
